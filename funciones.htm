function mcd(a, b) {  // Halla el máximo común divisor entre a y b
  return b == 0 ? a : mcd(b, a % b);
}

function deg2PIrad(g) {  //  conversion de grados sexagesimales a π radianes
  if (g == Math.trunc(g)) {
    cd = mcd(g, 180);
    return [g / cd, 180 / cd];
  } else if (Math.trunc(180/g) == 180/g) {
    return [1, 180/g];
  } else {
    return [g/180];  
  }
}
